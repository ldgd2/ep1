  seleccionarTallerDetalle(t: any) {
    this.tallerSeleccionado = { ...t };
    this.activeTab = 'taller-detalle';
    this.searchTermEspecialidades = '';
  }

  cerrarDetalleTaller() {
    this.activeTab = 'talleres';
    this.tallerSeleccionado = null;
    this.loadTalleres();
  }

  async toggleEspecialidad(taller: any, especialidadId: number) {
    const idsActuales = taller.especialidades?.map((e: any) => e.id) || [];
    let nuevosIds: number[] = [];

    if (idsActuales.includes(especialidadId)) {
      nuevosIds = idsActuales.filter((id: number) => id !== especialidadId);
    } else {
      nuevosIds = [...idsActuales, especialidadId];
    }

    try {
      const updated = await firstValueFrom(this.api.put<any>(`/talleres/${taller.cod}`, {
        especialidades: nuevosIds
      }));
      taller.especialidades = updated.especialidades;
      
      if (this.tallerSeleccionado && this.tallerSeleccionado.cod === taller.cod) {
         this.tallerSeleccionado.especialidades = updated.especialidades;
      }
    } catch (e) {
      alert("Error al actualizar especialidades del taller. Posiblemente por política CORS o sesión expirada.");
    }
  }
